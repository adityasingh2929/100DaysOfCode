FLEX:

        Center Properties:
            container-properties:
                1. justify-content: (center/etc.)  //aligns one/multiple lines acc. to main axis
                2. align-item: (center/etc.)       // aligns one line acc. to cross axis
                3. align-content: (center/etc.)    // aligns multiple lines acc. to cross axis
            item-properties:
                1. justify-self: (center/etc.)   // aligns one item acc. to main axis
                2. align-self: (center/etc.)     // aligns multiple items acc. to cross axis

    



    
GRID:

        Center Properties:
            container-properties:
                1. justify-content: (center/etc.) // aligns the whole grid on basis of the container (horizontally)
                2. align-content: (center/etc.)   // aligns the whole grid on basis of the container (vertically)
                3. justify-items: (center/etc.)   // aligns the content of 'ALL' the grid areas acc. to their grid areas (horizontally)
                4. align-items: (center/etc.)     // aligns the content of 'ALL' the grid areas acc. to their grid areas (vertically)
            item-properties:
                1. justify-self: (center/etc.) (this aligns one item with respect to its individual grid area, horizontally)
                2. align-self: (center/etc.)   (this aligns one item with respect to its individual grid area, vertically)
